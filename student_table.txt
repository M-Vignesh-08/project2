CREATE TABLE student
(
reg_no VARCHAR(20) PRIMARY KEY,
name VARCHAR(30),
department VARCHAR(20),
age INT,
address VARCHAR(50)
);

CREATE TABLE teacher(
staff_id INT PRIMARY KEY,
staff_name VARCHAR(20),
Department VARCHAR(20),
Date_of_join DATE,
salary INT
);

CREATE TABLE department(
dep_id INT PRIMARY KEY,
dep_name VARCHAR(20),
Stu_count INT,
HOD VARCHAR(20)
);
